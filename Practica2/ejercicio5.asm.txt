ORG 1000H
MSJ DB "INGRESE UN NUMERO:"
FAILMSG DB "CARACTER NO VALIDO."
FIN DB ?

ORG 1500H
NUM DB ?

ORG 3000H
ES_NUM: MOV DH, 30H
LOOP: CMP [BX], DH
JZ VERDADERO
INC DH
CMP DH, 40H
JZ FINSUB
JNZ LOOP
VERDADERO: MOV CH, 0FFH
JMP FINSUB
FALSO: MOV CH, 00H
FINSUB: RET

ORG 2000H
MOV BX, OFFSET MSJ
MOV AL, OFFSET FAILMSG-OFFSET MSJ
INT 7
MOV BX, OFFSET NUM
INT 6
MOV AL, 1
MOV DL, 0     ;ACA GUARDO SI ES O NO
CALL ES_NUM
CMP CH, 00H
JZ ERROR
JMP IMPRIME

ERROR:MOV AL, OFFSET FIN-OFFSET FAILMSG
MOV BX, OFFSET FAILMSG
INT 7
JMP TERMINA

IMPRIME: INT 7
MOV CL, NUM

TERMINA: INT 0
END
