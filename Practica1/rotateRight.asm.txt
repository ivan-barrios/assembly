;Memoria de datos
ORG 1000H
NUM DB 5H
NUMPOS DB 2H

;Subrutina ROTARIZ
ORG 3000H
ROTARIZ: CMP CL, 0
JZ FIN
CMP DL, 0
JZ FIN
NOTERMINO: ADD DL, DL
JNC NOCARRY
INC DL
NOCARRY: DEC CL
JNZ NOTERMINO
FIN: RET

;Subrutina ROTARDER
ORG 4000H
ROTARDER: MOV CL, CH
CALL ROTARIZ
RET 

;Programa principal
ORG 2000H
MOV CL, NUMPOS
MOV CH, 8H
SUB CH, CL   ;Cantidad de veces a rotar
MOV DL, NUM
CALL ROTARDER
HLT
END
