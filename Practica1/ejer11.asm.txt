ORG 1000H
STRING DB "IVANCITA"
LASTONE DB 00H
CHAR DB "A"
REEMP DB "X"

ORG 3000H
CONREMPCAR: MOV BX, DX
CMP BYTE PTR [BX], 00H
JZ FIN
LOOP: NOP
CMP [BX], AL
JNZ DIFF
INC CL
MOV [BX], AH
DIFF: INC BX
CMP BYTE PTR [BX], 00H
JNZ LOOP
FIN: RET

ORG 2000H
MOV AL, CHAR
MOV AH, REEMP
MOV CL, 0
MOV DX, OFFSET STRING
CALL CONREMPCAR
HLT
END
