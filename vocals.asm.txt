ORG 1000H
VOCALS DB "AaEeIiOo"
STRING DB "ivan"

ORG 3000H
;Se necesita: DL, BX, CX, AH
ES_VOCAL: PUSH BX
PUSH A
CMP DL, [BX]     
JZ CUMPLE
INC BX
DEC AH
JZ NOCUMPLE
JMP ES_VOCAL
CUMPLE: NOP
MOV DH, 0FFH
JMP FIN
NOCUMPLE: MOV DH, 00H
FIN: POP BX
RET

;CX --> DIRECCION DE STRING
;AL --> CONTADOR
;DH --> Tiene o no vocal
ORG 4000H
VOCALES: PUSH BX
PUSH CX
PUSH DX
MOV DX, CX
MOV BX, OFFSET VOCALS    ;BX --> direc vocals
SUB DX, BX
MOV AH, DL               ;AH --> cantidad vocals

LOOP: MOV DL, [CX]       ;DL --> caracter
CALL ES_VOCAL
CMP DH, 0FFH
JZ CONTAR

CONTAR: INC AL
INC CX

POP DX
POP CX
POP BX
RET

ORG 2000H
MOV AL, 00H              ;AL contador
MOV CX, OFFSET STRING
CALL VOCALES
HLT
END
